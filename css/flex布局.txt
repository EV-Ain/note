flex属性值
flex-direction:row;//顺序排列子元素，从左到右排列 默认
flex-direction:row-reverse;//顺序排列子元素，从右向左排列
flex-direction:column;//顺序排列子元素，从上到下排列
flex-direction:column-reverse;//顺序排列子元素，从下到上排列

//设置主轴子元素排列方向
just-content:flex-start;//头部开始从左到右排列 默认
just-content:flex-end;//从尾部开始排列

just-content:center;//在主轴居中排列 左右居中
just-content:space-between;//先两边贴边然后平分剩余空间
just-content:space-around;//平分剩余空间

//flex换行
flex-wrap:nowrap;//默认子元素不换行 如果装不下会缩小子元素
flex-wrap:wrap;//子元素换行

//设置侧轴 单行设置
align-items:flex-start;//
align-items:stretch;//子元素填满父元素 子元素不要给高度
align-items:center;//垂直居中   加上just-content:center上下左右居中
